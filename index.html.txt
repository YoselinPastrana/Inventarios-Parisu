<script>

let inventario = JSON.parse(localStorage.getItem("inventarioParisu")) || [];
let historial = JSON.parse(localStorage.getItem("historialParisu")) || [];

function guardar(){
    localStorage.setItem("inventarioParisu", JSON.stringify(inventario));
    localStorage.setItem("historialParisu", JSON.stringify(historial));
}

function registrarMovimiento(){

    let tipo = document.getElementById("tipoMovimiento").value;
    let manga = document.getElementById("manga").value;
    let corte = document.getElementById("corte").value;
    let talla = document.getElementById("talla").value;
    let color = document.getElementById("color").value.trim();
    let cantidad = parseInt(document.getElementById("cantidad").value);

    if(!color || isNaN(cantidad)) return alert("Completa todos los campos");

    let producto = inventario.find(p =>
        p.manga==manga &&
        p.corte==corte &&
        p.talla==talla &&
        p.color.toLowerCase()==color.toLowerCase()
    );

    if(tipo==="entrada"){
        if(producto){
            producto.stock += cantidad;
        } else {
            inventario.push({manga, corte, talla, color, stock:cantidad});
        }
    }

    if(tipo==="salida"){
        if(!producto || producto.stock < cantidad){
            return alert("No hay suficiente stock disponible");
        }
        producto.stock -= cantidad;
    }

    historial.push({
        fecha:new Date().toLocaleString(),
        tipo,
        manga,
        corte,
        talla,
        color,
        cantidad
    });

    guardar();
    mostrarInventario();
    mostrarHistorial();

    document.getElementById("color").value="";
    document.getElementById("cantidad").value="";
}

function mostrarInventario(){
    let tabla = document.getElementById("tablaInventario");
    tabla.innerHTML="";
    let total=0;

    inventario.forEach(item=>{
        total+=item.stock;
        tabla.innerHTML+=`
        <tr class="${item.stock<=5?'low':''}">
            <td>${item.manga}</td>
            <td>${item.corte}</td>
            <td>${item.talla}</td>
            <td>${item.color}</td>
            <td>${item.stock}</td>
        </tr>
        `;
    });

    document.getElementById("totalPiezas").innerText=total;
}

function mostrarHistorial(){
    let tabla=document.getElementById("tablaHistorial");
    tabla.innerHTML="";

    historial.slice().reverse().forEach(m=>{
        tabla.innerHTML+=`
        <tr>
            <td>${m.fecha}</td>
            <td class="${m.tipo==='entrada'?'mov-entrada':'mov-salida'}">
                ${m.tipo.toUpperCase()}
            </td>
            <td>${m.manga}</td>
            <td>${m.corte}</td>
            <td>${m.talla}</td>
            <td>${m.color}</td>
            <td>${m.cantidad}</td>
        </tr>
        `;
    });
}

function exportarCSV(){
    let csv="Fecha,Tipo,Manga,Corte,Talla,Color,Cantidad\n";
    historial.forEach(m=>{
        csv+=`${m.fecha},${m.tipo},${m.manga},${m.corte},${m.talla},${m.color},${m.cantidad}\n`;
    });

    let blob=new Blob([csv],{type:"text/csv"});
    let url=URL.createObjectURL(blob);
    let a=document.createElement("a");
    a.href=url;
    a.download="historial_parisu.csv";
    a.click();
}

mostrarInventario();
mostrarHistorial();
